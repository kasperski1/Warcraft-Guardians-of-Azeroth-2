# Feudal Raiding on Land
tradition_might_is_right = {
	category = societal

	layers = {
		0 = martial
		1 = mediterranean
		4 = fight.dds
	}
	
	is_shown = {
		heritage_gives_sea_raiding_trigger = no
		NOT = { scope:character.faith = { like_pacifism_faith_trigger = yes } }
	}
	
	can_pick = {
		custom_tooltip = {
			text = culture_not_pacifist_trigger_desc
			culture_not_pacifistic_trigger = yes
		}
		
		custom_tooltip = {
			text = cannot_have_tradition_isolationist_desc
			NOT = { culture_tradition:tradition_isolationist = { is_in_list = traits } }
		}
		
		custom_tooltip = {
			text = cannot_have_tradition_practiced_pirates
			NOT = { culture_tradition:tradition_practiced_pirates = { is_in_list = traits } }
		}
	}
	
	parameters = {
		culture_can_raid_even_if_feudal = yes
		prowess_traits_more_valued = yes
	}
	county_modifier = {
		monthly_county_control_change_factor = -0.2
	}

	character_modifier = {
		knight_effectiveness_mult = 0.15
		opinion_of_liege = -10
		cultural_head_fascination_mult = -0.1
		accolade_glory_gain_mult = 0.1
		ai_boldness = 25
	}
	
	cost = {
		prestige = {
			add = {
				value = tradition_base_cost
				desc = BASE
				format = "BASE_VALUE_FORMAT"
			}
			# Ethos requirement.
			if = {
				limit = {
					NOR = {
						culture_pillar:ethos_bellicose = { is_in_list = traits }
						culture_pillar:ethos_stoic = { is_in_list = traits }
					}
				}
				add = {
					value = tradition_incompatible_ethos_penalty
					desc = not_bellicose_or_stoic_desc
				}
			}
			multiply = tradition_replacement_cost_if_relevant
		}
	}
	
	ai_will_do = {
		value = 100
		if = {
			limit = {
				NOR = { 
					culture_pillar:ethos_bellicose = { is_in_list = traits }
					culture_pillar:ethos_stoic = { is_in_list = traits }
				}
			}
			multiply = 0
		}
	}
}

tradition_seducers = {
	category = ritual

	layers = {
		0 = intrigue
		1 = indian
		4 = apple.dds
	}
	
	parameters = {
		prestige_from_seduction_schemes = yes
		seduce_success_chance_per_prestige_level = yes
		lustful_trait_more_common = yes
		chaste_trait_less_common = yes
		seducer_trait_prestige = yes
		#seducers_same_culture_opinion = yes
		seduce_scheme_penalty = yes
		seduce_scheme_bonuses = yes
		beauty_traits_more_valued = yes
		ugly_traits_looked_down_upon = yes
		beauty_traits_more_common = yes
	}
	character_modifier = {
		seduce_scheme_power_mult = 0.15
	}
	cost = {
		prestige = {
			add = {
				value = tradition_base_cost
				desc = BASE
				format = "BASE_VALUE_FORMAT"
			}
			if = {
				limit = {
					NOR = {
						culture_pillar:ethos_courtly = { is_in_list = traits }
						culture_pillar:ethos_communal = { is_in_list = traits }
					}
				}
				add = {
					value = tradition_incompatible_ethos_penalty
					desc = not_courtly_or_communal_desc
				}
			}
			if = {
				limit = {
					trigger_if = {
						limit = {
							scope:character = {
								is_ai = no
							}
						}
						NOT = {
							any_ruler = {
								count >= 5
								culture = prev
								primary_title.tier >= tier_county
								has_trait = seducer
							}
						}
					}
					trigger_else = {
						always = no
					}
				}
				add = {
					value = tradition_unfulfilled_criteria_penalty
					desc = rulers_with_seducer_lifestyle_desc
				}
			}
			
			multiply = tradition_replacement_cost_if_relevant
		}
	}
	
	ai_will_do = {
		value = 100
		if = {
			limit = {
				NOT = {
					scope:character = {
						OR = {
							has_trait = seducer
							has_trait = lustful
						}
					}
				}
			}
			multiply = 0.5
		}
	}
}
